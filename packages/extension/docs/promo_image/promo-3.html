my-agent.js
// Your web page can control the browser
// via the Page Agent Extension API
 
const result = await window.PAGE_AGENT_EXT.execute(
  "Book meeting room A for
   tomorrow 2pm on the calendar",
  {
    model: "gpt-4o",
    apiKey: "sk-...",
    onActivity: (a) =>
      console.log(a.description)
  }
)
 
if (result.success) {
  showToast("Room booked!")
}
